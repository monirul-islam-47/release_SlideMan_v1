    5. Theme Loading (src/pptx_manager/theme.py):
      from PySide6.QtCore import QFile, QTextStream, QIODevice
from PySide6.QtWidgets import QApplication

_current_theme = "dark" # Default

def load_stylesheet(theme_name: str) -> str:
    qss_path = f":/qss/{theme_name}.qss"
    file = QFile(qss_path)
    if file.open(QIODevice.ReadOnly | QIODevice.Text):
        stream = QTextStream(file)
        stylesheet = stream.readAll()
        file.close()
        return stylesheet
    return ""

def apply_theme(app: QApplication, theme_name: str):
    global _current_theme
    stylesheet = load_stylesheet(theme_name)
    if stylesheet:
        app.setStyleSheet(stylesheet)
        _current_theme = theme_name

def get_current_theme() -> str:
    return _current_theme

DB_SCHEMA_VERSION = 1

class Database:
    def __init__(self, db_path: Path):
        self.db_path = db_path
        self._conn = None

    def connect(self):
        self._conn = sqlite3.connect(self.db_path)
        self._conn.row_factory = sqlite3.Row # Access columns by name
        self._conn.execute("PRAGMA foreign_keys = ON;")
        self._initialize_schema()

    def close(self):
        if self._conn:
            self._conn.close()
            self._conn = None

    def _initialize_schema(self):
        current_version = self._conn.execute("PRAGMA user_version;").fetchone()[0]
        if current_version < DB_SCHEMA_VERSION:
            if current_version == 0: # Initial creation
                self._create_tables()
                self._conn.execute(f"PRAGMA user_version = {DB_SCHEMA_VERSION};")
            # else: Add migration logic here for future versions
            self._conn.commit()

    def _create_tables(self):
